This is the sample script for Creation of Storage Account


-- main.tf

#Define providers used
provider "azurerm" {
  version = "=2.51.0"
  features {} #this is required for v2 of the provider even if empty or paln with fail
}

#Data section
data "azurerm_resource_group" "DevResourceGroup" {
  name = "DevResourceGroup"
}

#Resource section
resource "azurerm_storage_account" "GuganSA" {
  name                     = "gugansa"
  resource_group_name      = data.azurerm_resource_group.DevResourceGroup.name
  location                 = "Central US"
  account_kind             = "StorageV2"
  account_tier             = "Standard"
  account_replication_type = var.replicationType
}


-- variable.tf

variable "replicationType" {
  description = "The type of replication"
  type        = string
  default     = "LRS"
}

-- output.tf

output "Storage_Account_Name" {
  value = azurerm_storage_account.GuganSA.name
}
